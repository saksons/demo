# На маршрутизаторах сконфигурируйте статическую трансляцию портов
Создаём 2 object group для порта и для сервера  
Создаём zone pair из UNTRUSTED в TRUSTED  
Создаём dnat НО ДЛЯ dest-addr ставим WAN_NET (т.е. белый адрес)  

> **Пробросьте порт 8080 в порт приложения testapp BR-SRV на маршрутизаторе BR-RTR, для обеспечения работы приложения testapp извне**

```bash
#BR-RTR
object-group service TESTAPP
port-range 8080
exit

object-group network SERVER
ip address-range 192.168.255.2
exit

security zone-pair UNTRUST TRUST
rule 1
action permit
match protocol tcp
match destionation-port object-group TESTAPP
match destionation-address object-group SERVER
enable
exit
exit

nat destination
ruleset DNAT
from zone UNTRUST
pool SERVER
ip address 192.168.255.2
exit
rule 1
match protocol tcp
match destination-address object-group WAN_NET
match destination-port object-group TESTAPP
action destination-nat pool SERVER
enable
end
commit
confirm

```
> **Пробросьте порт 8080 в порт веб приложения на HQ-SRV на маршрутизаторе HQ-RTR, для обеспечения работы веб приложения извне**
```bash
# HQ-RTR
object-group network SERVER
ip address-range 192.168.100.2
exit
object-group service APACHE
port-range 80
exit

security zone-pair UNTRUST TRUST
rule 1
action permit
match protocol tcp
match destination-address object-group SERVER
match destination-port object-group APACHE
enable
exit
exit

nat destination
pool SERVER
ip address 192.168.100.2
exit
ruleset DNAT
rule 1
action destination-nat pool SERVER
match protocol tcp
match destination-address object-group WAN_NET
match destination-port object-group APACHE
enable
end
commit
confirm
```
