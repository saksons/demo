>**На всех устройствах необходимо сконфигурировать IPv4:**
>**• IP-адрес должен быть из приватного диапазона, в случае, если сеть локальная, согласно RFC1918**
>**• Локальная сеть в сторону HQ-SRV(VLAN 100) должна вмещать не более 32 адресов**
>**• Локальная сеть в сторону HQ-CLI(VLAN 200) должна вмещать не менее 16 адресов**
>**• Локальная сеть для управления (VLAN 999) должна вмещать не более 8 адресов**
>**• Локальная сеть в сторону BR-SRV должна вмещать не более 16 адресов**

```bash
# HQ-RTR
sh int stat
configure
interface gi1/0/x # to hq-srv
ip firewall disable
ip address 192.168.100.1/27
exit
interface gi1/0/y # to hq-cli
ip firewall disable
ip address 192.168.200.1/28
exit
exit
commit
confirm

# BR-RTR
configure
interface gi1/0/z # to br-srv
ip firewall disable
ip address 192.168.255.1/28
exit
exit
commit
confirm


#HQ-SRV
nmcli con show
nmcli con mod {con_name} ipv4.method manual ipv4.address 192.168.100.2/27 ipv4.gateway 192.168.100.1 ipv4.dns "77.88.8.8" # to hq-rtr
nmcli con up {con_name}

#HQ-CLI
nmcli con show
nmcli con mod {con_name} ipv4.method manual ipv4.address 192.168.200.2/28 ipv4.gateway 192.168.200.1 ipv4.dns "77.88.8.8" # to hq-rtr
nmcli con up {con_name}

#BR-SRV
nmcli con show
nmcli con mod {con_name} ipv4.method manual ipv4.address 192.168.255.2/28 ipv4.gateway 192.168.200.1 ipv4.dns "77.88.8.8" # to br-rtr
nmcli con up {con_name}
```
